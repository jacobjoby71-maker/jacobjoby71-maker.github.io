GOOGLE APPS SCRIPT — SETUP CODE

Step 1: Open Google Sheets, create a new sheet named Contributions.

Step 2: Go to Extensions → Apps Script, paste the code below, save, then click Deploy → New Deployment → Web App. Set access to "Anyone". Copy the URL into the field above.

Step 3: Add the same URL to all three public forms (website, Facebook, WhatsApp) in the ENDPOINT constant at the top of each file. 
const SHEET_ID = 'YOUR_GOOGLE_SHEET_ID_HERE';
const SHEET_NAME = 'Contributions';

function doPost(e) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      sheet.appendRow(['Timestamp','Channel','Name','Ref No','Prayer Group',
        'Phone','Email','Date','Voluntary','Birthday','Ingathering',
        'Christmas','Wedding Anniv','Twelve Contrib','Suvisesha Sangam',
        'Recessa','Maramon Conv','Self Denial','Poor Fund','Tithe',
        'Family Dedication','Harvest Festival','Financial Support',
        'Other 1','Other 2','Other 3','Total CAD','Notes','Trustee']);
    }
    const d = JSON.parse(e.postData.contents);
    const cats = ['Voluntary Contribution','Birthday','Ingathering',
      'Christmas','Wedding Anniversary','Twelve Contribution',
      'Suvisesha Sangam Anduvary','Recessa','Maramon Convention',
      'Self Denial','Poor Fund','Tithe','Family Dedication Day',
      'Harvest Festival','Financial Support'];
    const row = [
      new Date(), d.channel||'', d.name||'', d.ref||'',
      d.prayer||'', d.phone||'', d.email||'', d.date||'',
      ...cats.map(c => d.contributions?.[c] || 0),
      d.other1||0, d.other2||0, d.other3||0,
      d.total||0, d.notes||'', d.trustee||''
    ];
    sheet.appendRow(row);
    return output({success:true, id: Utilities.getUuid()});
  } catch(err) {
    return output({success:false, error: err.message});
  }
}

function doGet(e) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) return output([]);
  const rows = sheet.getDataRange().getValues();
  const headers = rows[0];
  const data = rows.slice(1).map(r => {
    const obj = {};
    headers.forEach((h,i) => obj[h] = r[i]);
    return obj;
  });
  return output(data);
}

function output(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
